FORM get_data .
  SELECT *  INTO CORRESPONDING FIELDS OF TABLE gt_rent FROM ztrent_b02
    INNER JOIN ztmember_b02 ON ztrent_b02~mb_id = ztmember_b02~mb_id
    WHERE ztrent_b02~mb_id IN so_id AND ztmember_b02~mb_st = 'A'
    and ztrent_b02~return_st = 'A'.
  SELECT * FROM ztbook_b02 INTO CORRESPONDING FIELDS OF TABLE gt_book.
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      domname         = 'ZD_RTST_B02'
      text            = 'X'
    TABLES
      values_tab      = gt_itab
    EXCEPTIONS
      no_values_found = 1
      OTHERS          = 2.
  IF sy-subrc <> 0.
    MESSAGE e000(zb02) WITH '도메인이 잘못되었습니다'.
  ENDIF.

ENDFORM.

FORM make_data .
  LOOP AT gt_rent INTO gs_rent.
    DATA: lv_date TYPE dats.
    DATA: lv_date1 TYPE datS.
    CALL FUNCTION 'RP_CALC_DATE_IN_INTERVAL'
      EXPORTING
        date      = sy-datum
        days      = '1'
        months    = '0'
        signum    = '+'
        years     = '0'
      IMPORTING
        calc_date = lv_date.
    CALL FUNCTION 'RP_CALC_DATE_IN_INTERVAL'
      EXPORTING
        date      = sy-datum
        days      = '2'
        months    = '0'
        signum    = '+'
        years     = '0'
      IMPORTING
        calc_date = lv_date1.



    READ TABLE gt_book INTO gs_book WITH KEY book_id = gs_rent-book_id.
    gs_rent-book_nm = gs_book-book_nm.
    READ TABLE gt_itab INTO gs_itab WITH KEY domvalue_l = gs_rent-return_st.
    gs_rent-rdtext = gs_itab-ddtext.
    IF gs_rent-return_dt < sy-datum.
      gs_rent-light = '1'.
    ELSEIF gs_rent-return_dt = lv_date.
      gs_rent-light = '2'.
    ELSEIF gs_rent-return_dt >= lv_date1.
      gs_rent-light = '3'.
    ENDIF.
    MODIFY gt_rent FROM gs_rent TRANSPORTING book_nm rdtext light.
  ENDLOOP.

ENDFORM.
